```python
from __future__ import annotations

import json
from collections.abc import Awaitable, Callable
from typing import Literal

from pydantic import BaseModel

__all__ = [
    "AllScansComplete",
    "ConfigGenerated",
    "ConversationStart",
    "FrontDoorMessage",
    "PasseleweMessage",
    "ScanCallback",
    "ScanComplete",
    "ScanStart",
    "ScanUpdate",
    "UserMessage",
    "parse_client_message",
    "parse_server_message",
]


class FrontDoorMessage(BaseModel):
    """Base for all Front Door WebSocket messages."""

    model_config = {"frozen": True}


class ScanStart(FrontDoorMessage):
    type: Literal["scan_start"] = "scan_start"
    panels: list[str]


class ScanUpdate(FrontDoorMessage):
    type: Literal["scan_update"] = "scan_update"
    panel: str
    label: str
    value: str
    detail: str = ""


ScanCallback = Callable[[ScanUpdate], Awaitable[None]]
"""Async callback for streaming scan events."""


class ScanComplete(FrontDoorMessage):
    type: Literal["scan_complete"] = "scan_complete"
    panel: str
    item_count: int


class AllScansComplete(FrontDoorMessage):
    type: Literal["all_scans_complete"] = "all_scans_complete"
    total_items: int


class ConversationStart(FrontDoorMessage):
    type: Literal["conversation_start"] = "conversation_start"


class PasseleweMessage(FrontDoorMessage):
    type: Literal["passelewe_message"] = "passelewe_message"
    text: str
    subtype: Literal["narration", "question", "reflection"]


class ConfigGenerated(FrontDoorMessage):
    type: Literal["config_generated"] = "config_generated"
    config_toml: str
    annotations: dict[str, str]


class UserMessage(FrontDoorMessage):
    type: Literal["user_message"] = "user_message"
    text: str


_SERVER_TYPES: dict[str, type[FrontDoorMessage]] = {
    "scan_start": ScanStart,
    "scan_update": ScanUpdate,
    "scan_complete": ScanComplete,
    "all_scans_complete": AllScansComplete,
    "conversation_start": ConversationStart,
    "passelewe_message": PasseleweMessage,
    "config_generated": ConfigGenerated,
}

_CLIENT_TYPES: dict[str, type[FrontDoorMessage]] = {
    "user_message": UserMessage,
}


def _parse(raw: str, registry: dict[str, type[FrontDoorMessage]]) -> FrontDoorMessage:
    try:
        data = json.loads(raw)
    except json.JSONDecodeError as exc:
        msg = f"Invalid JSON: {exc}"
        raise ValueError(msg) from exc
    msg_type = data.get("type")
    cls = registry.get(msg_type) if msg_type else None
    if cls is None:
        msg = f"Unknown message type: {msg_type!r}"
        raise ValueError(msg)
    return cls.model_validate(data)


def parse_client_message(raw: str) -> FrontDoorMessage:
    return _parse(raw, _CLIENT_TYPES)


def parse_server_message(raw: str) -> FrontDoorMessage:
    return _parse(raw, _SERVER_TYPES)
```

### `src/bonfire/onboard/orchestrator.py`

```python
from __future__ import annotations

import asyncio
import logging
from typing import TYPE_CHECKING

from bonfire.onboard.protocol import (
    AllScansComplete,
    FrontDoorMessage,
    ScanComplete,
    ScanStart,
    ScanUpdate,
)
from bonfire.onboard.scanners import (
    claude_memory,
    cli_toolchain,
    git_state,
    mcp_servers,
    project_structure,
    vault_seed,
)

if TYPE_CHECKING:
    from collections.abc import Awaitable, Callable
    from pathlib import Path
    from types import ModuleType

__all__ = ["run_scan"]

_log = logging.getLogger(__name__)


def _get_scanners() -> list[tuple[str, ModuleType]]:
    return [
        ("project_structure", project_structure),
        ("cli_toolchain", cli_toolchain),
        ("claude_memory", claude_memory),
        ("git_state", git_state),
        ("mcp_servers", mcp_servers),
        ("vault_seed", vault_seed),
    ]


async def run_scan(
    project_path: Path,
    emit: Callable[[FrontDoorMessage], Awaitable[None]],
) -> int:
    scanners = _get_scanners()
    panel_names = [name for name, _ in scanners]
    await emit(ScanStart(panels=panel_names))

    tasks = [_run_one(panel, module, project_path, emit) for panel, module in scanners]
    results: list[int] = await asyncio.gather(*tasks)

    total = sum(results)
    await emit(AllScansComplete(total_items=total))
    return total


async def _run_one(
    panel: str,
    module: ModuleType,
    project_path: Path,
    emit: Callable[[FrontDoorMessage], Awaitable[None]],
) -> int:
    async def _narrow_emit(event: ScanUpdate) -> None:
        await emit(event)

    try:
        count = await module.scan(project_path, _narrow_emit)
    except Exception:
        _log.exception("Scanner %s failed", panel)
        count = 0

    await emit(ScanComplete(panel=panel, item_count=count))
    return count
```

## Appendix — v1 surprises (scout notes)

1. **`PANEL` vs `_PANEL` constant naming inconsistency.** v1 mixes: `cli_toolchain.py:_PANEL`, `mcp_servers.py:_PANEL`, vs `git_state.py:PANEL`, `claude_memory.py:PANEL`, `vault_seed.py:PANEL`, `project_structure.py:PANEL`. Both are module-private (no test imports them by name). Conservative Warrior keeps verbatim. Innovative Warrior may consolidate to `_PANEL` everywhere — but verify no test references break.

2. **`_Q2_TEXT` em-dash escape sequence.** v1 conversation.py:36 uses `—` (em-dash). v0.1 ruff may flag literal em-dashes (RUF001/RUF002). Keep `—` to suppress. The v0.1 codebase is mixed — some files use literal em-dashes, some use escapes. Match v1 verbatim per §D5.

3. **Race condition in `flow.py:90-95`.** The `server.on_message = on_message` is set AFTER `conversation.start(...)` to avoid early-message races. v1 explicitly documents this (line 93-95 comment). Port verbatim. Do NOT "clean up" by setting the callback before start.

4. **Hang behavior in `flow.py:99-100`.** If browser closes mid-conversation, `conversation_done.wait()` hangs until Ctrl-C. v1 documents this is "Acceptable for v1 — a future improvement could race against shutdown_event." Port verbatim. Do NOT add a race here — that's a follow-up ticket.

5. **`_run_one`'s `_narrow_emit` closure.** v1 orchestrator.py:99-102 wraps the emit callable in a `_narrow_emit` closure that just forwards. This looks redundant (why not pass `emit` directly?). But the type signature differs: orchestrator emits `FrontDoorMessage`, scanner expects `Callable[[ScanUpdate], Awaitable[None]]`. The closure is a type-narrowing identity function. Port verbatim.

6. **`broadcast` snapshot with `return_exceptions=True`.** v1 server.py:113 passes `return_exceptions=True` to `asyncio.gather` so one bad client doesn't sink the broadcast. The deferred cli_audit `test_broadcast_tolerates_send_failure` pins this. Port verbatim. Do NOT remove `return_exceptions=True` — it's load-bearing.

7. **`_load_html` reads bytes once at start time.** v1 server.py:86 calls `self._html = _load_html()` inside `start()`, NOT at `__init__()` or class-load time. Port verbatim — late binding lets the file path resolve correctly under `importlib.resources` even if working directory changes.

8. **`_ws_handler` finally block sets `_shutdown_event`.** v1 server.py:183-186: `self._clients.discard(websocket); if self._had_clients and not self._clients: self._shutdown_event.set()`. The `_had_clients` gate ensures the event doesn't fire on a server that never had a client connect. The deferred cli_audit `test_shutdown_event_starts_unset` and `test_had_clients_starts_false` pin both initial values. Port verbatim.

9. **Q1 default-fallback profile values.** v1 conversation.py:144-149 sets a "default: result-oriented" branch when no Q1 keywords match. The reflection text is REUSED from the result branch (line 134-138) — same string. This is intentional: short / generic answers default to the most common archetype. Port verbatim. The cli_audit test `test_q1_default_path_*` (deferred to D-FT A) probes this.

10. **`_analyze_q3` uses `startswith` for keyword matching.** v1 conversation.py:315-322 does `any(w.strip(...).startswith(k) for k in speed_kw)` — partial-word matching. The deferred cli_audit `test_knowing_should_not_match_understand` documents that "knowing" matches "know" (in understand_kw). This is a known minor false-positive; port verbatim. A future ticket could swap to whole-word matching.

11. **`_ensure_complete_profile` defaults are 6 keys.** v1 conversation.py:451-463: companion_mode, goal_visibility, energy_type, attention_topology, uncertainty_orientation, pain_point. The deferred cli_audit `test_complete_profile_has_all_six_dimensions` pins this. Port verbatim.
